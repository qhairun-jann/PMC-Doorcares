responsive 4 button row 


                <section class="row">
                        <div class="col-12 col-md-6 col-xl-3 text-center" v-for="(btn,index) in buttons" :key="index">
                            <button class="btn btn-outline-info btn-lg my-2" v-bind:class="{ active: index === btnActive }"
                            @click="handleClick" :id="index" style="width: 70%;">
                             {/{ btn }/}
                            </button>
                        </div>
                </section>
                


                <script src="https://cdn.jsdelivr.net/npm/vue@2"></script>
                <script>
                var app = new Vue({
                    el: '#vue-app',
                    data: {
                        btnActive: 0,
                        buttons: ['All', 'New', 'In Progress', 'Resolved'],
                    },
                    methods: {
                        handleClick(event) {
                            document.getElementById(this.btnActive).classList.remove('active')
                            this.btnActive = event.target.id
                            event.target.classList.add('active')
                        }
                    }
                })
                </script>
